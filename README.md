```cpp
#include <SoftwareSerial.h>

// Bluetooth pins
#define BT_TX 10
#define BT_RX 11
SoftwareSerial bluetooth(BT_TX, BT_RX);

// Front ultrasonic sensor
#define FRONT_TRIG 7
#define FRONT_ECHO 6

// Down ultrasonic sensor
#define DOWN_TRIG 5
#define DOWN_ECHO 4

// Vibration motor
#define VIBRATION_PIN 9

// Device state
bool deviceEnabled = false;
int frontThreshold = 100; // cm
String downSensitivity = "medium";

// Sensitivity settings
int downThresholdLow = 50;    // cm
int downThresholdMedium = 30; // cm
int downThresholdHigh = 15;   // cm

void setup() {
  Serial.begin(9600);
  bluetooth.begin(9600);

  // Ultrasonic sensors
  pinMode(FRONT_TRIG, OUTPUT);
  pinMode(FRONT_ECHO, INPUT);
  pinMode(DOWN_TRIG, OUTPUT);
  pinMode(DOWN_ECHO, INPUT);

  // Vibration motor
  pinMode(VIBRATION_PIN, OUTPUT);
  digitalWrite(VIBRATION_PIN, LOW);

  Serial.println("Arduino Bluetooth Sensor Ready");
}

void loop() {
  // Read Bluetooth commands
  if (bluetooth.available()) {
    String command = bluetooth.readStringUntil('\n');
    command.trim();
    processCommand(command);
  }

  // Send sensor data if device is enabled
  if (deviceEnabled) {
    sendSensorData();
  }

  delay(100);
}

void processCommand(String command) {
  Serial.print("Received: ");
  Serial.println(command);

  if (command == "TURN_ON") {
    deviceEnabled = true;
    bluetooth.println("DEVICE:ON");
  }
  else if (command == "TURN_OFF") {
    deviceEnabled = false;
    digitalWrite(VIBRATION_PIN, LOW);
    bluetooth.println("DEVICE:OFF");
  }
  else if (command.startsWith("SET_FRONT_THRESHOLD:")) {
    frontThreshold = command.substring(19).toInt();
    bluetooth.print("THRESHOLD_SET:");
    bluetooth.println(frontThreshold);
  }
  else if (command.startsWith("SET_DOWN_SENSITIVITY:")) {
    downSensitivity = command.substring(20);
  }
  else if (command == "TEST_VIBRATION") {
    digitalWrite(VIBRATION_PIN, HIGH);
    delay(500);
    digitalWrite(VIBRATION_PIN, LOW);
  }
  else if (command == "ALERT_MODE") {
    // Maximum vibration for alert
    analogWrite(VIBRATION_PIN, 255);
    delay(1000);
    digitalWrite(VIBRATION_PIN, LOW);
  }
}

void sendSensorData() {
  // Front sensor
  long frontDistance = getDistance(FRONT_TRIG, FRONT_ECHO);
  bluetooth.print("FRONT_DIST:");
  bluetooth.println(frontDistance);

  // Check front threshold
  if (frontDistance > 0 && frontDistance <= frontThreshold) {
    digitalWrite(VIBRATION_PIN, HIGH);
    delay(200);
    digitalWrite(VIBRATION_PIN, LOW);
  }

  // Down sensor
  long downDistance = getDistance(DOWN_TRIG, DOWN_ECHO);
  bluetooth.print("DOWN_DIST:");
  bluetooth.println(downDistance);

  // Check down sensitivity
  int threshold = getDownThreshold();
  if (downDistance > 0 && downDistance <= threshold) {
    digitalWrite(VIBRATION_PIN, HIGH);
    delay(150);
    digitalWrite(VIBRATION_PIN, LOW);
  }
}

long getDistance(int trigPin, int echoPin) {
  digitalWrite(trigPin, LOW);
  delayMicroseconds(2);
  digitalWrite(trigPin, HIGH);
  delayMicroseconds(10);
  digitalWrite(trigPin, LOW);

  long duration = pulseIn(echoPin, HIGH);
  long distance = duration * 0.034 / 2;
  return distance;
}

int getDownThreshold() {
  if (downSensitivity == "low") return downThresholdLow;
  if (downSensitivity == "high") return downThresholdHigh;
  return downThresholdMedium; // medium
}
```
